public class Graph {
	private final int V;
	privateList<Integer>[] adj;

	public Graph(int V) {
		this.V = V;
		adj = (List<Integer>[]) new ArrayList[V];
		for（int v = θ;v < V;v++）{
			adj[v] = new ArrayList<Integer>();
		}
	}

	public void addEdge(int v, int w){
		adj[v].add(w); adj[w].add(v);
	}

	public Iterable<Integer> adj(int v) {
		return adj[v];
	}
}
