public class IteratorOfIterators     {
	List<Iterator<Integer>> list;
	int wizpos;
	int lastHasNext;

	public IteratorOfIterators(List<Iterator<Integer>> a) {
		this.wizpos = 0;
		this.list = a;
		this.lastHasNext = list.size() - 1;

	@Override
	public boolean hasNext() {
			if (list.size() == 0) return false;
			while (true) { 
				if (list.get(wizpos).hasNext()) {
					lastHasNext = wizpos;
					return true;
				}
				if (wizpos == lastHasNext) {
					return false;
				}
				wizpos = (wizpos + 1) % list.size();
			}
	}

	@Override
	public Integer next() {
		if (!hasNext()) {
			throw new NoSuchElementException();
		}
		wizpos = (wizpos + 1) % list.size();
		return list.get(lastHasNext).next();
	}
}
